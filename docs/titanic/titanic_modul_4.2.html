EDA - Analiza danych z katastrofy Titanica

O Danych

Dane o pasaerach Titanica

Zbir danych zawiera informacje o pasaerach RMS Titanic, ktry zaton 15 kwietnia 1912 roku po zderzeniu z gr lodow. Dane obejmuj takie atrybuty jak klasa podry, wiek, pe, liczba rodzestwa/maonkw na pokadzie, liczba rodzicw/dzieci na pokadzie, cena biletu oraz miejsce zaokrtowania.

Zbir zawiera take informacj o tym, czy pasaer przey katastrof.

Titanic przewozi ponad 2,200 osb, z czego ponad 1,500 zgino, co czyni t katastrof jedn z najbardziej tragicznych w historii morskiej.

Kolumny:

  • pclass - Klasa biletu
  • survived - Czy pasaer przey katastrof
  • name - Imi i nazwisko pasaera
  • sex - Pe pasaera
  • age - Wiek pasaera
  • sibsp - Liczba rodzestwa/maonkw na pokadzie
  • parch - Liczba rodzicw/dzieci na pokadzie
  • ticket - Numer biletu
  • fare - Cena biletu
  • cabin - Numer kabiny
  • embarked - Port, w ktrym pasaer wszed na pokad (C = Cherbourg, Q = Queenstown, S = Southampton)
  • boat - Numer odzi ratunkowej
  • body - Numer ciaa (jeli pasaer nie przey i ciao zostao odnalezione)
  • home.dest - Miejsce docelowe

# Funkcja do obliczania statystyk przeycia
def calculate_survival_stats(df, condition):
    total_count = df[condition].shape[0]
    survived_count = df[condition & (df['survived'] == 1)].shape[0]
    died_count = total_count - survived_count
    survived_percent = (survived_count / total_count) * 100 if total_count > 0 else 0
    died_percent = (died_count / total_count) * 100 if total_count > 0 else 0
    return [total_count, survived_count, survived_percent, died_count, died_percent]

# Zdefiniowane warunki dla kadego parametru
conditions = {
    'pclass 1': df['pclass'] == 1,
    'pclass 2': df['pclass'] == 2,
    'pclass 3': df['pclass'] == 3,
    'male': df['sex'] == 'male',
    'female': df['sex'] == 'female',
    'boat /numer': df['boat'].notna(),
    'boat/NaN': df['boat'].isna(),
    'sibsp 0': df['sibsp'] == 0,
    'sibsp liczba': df['sibsp'] > 0,
    'parch 0': df['parch'] == 0,
    'parch liczba': df['parch'] > 0,
    'age do 10': df['age'] <= 10,
    'age >10 do 20': (df['age'] > 10) & (df['age'] <= 20),
    'age >20 do 40': (df['age'] > 20) & (df['age'] <= 40),
    'age > 40 do 60': (df['age'] > 40) & (df['age'] <= 60),
    'age > 60': df['age'] > 60
}

# Obliczenie statystyk dla kadego warunku
stats = {param: calculate_survival_stats(df, cond) for param, cond in conditions.items()}

# Utworzenie DataFrame ze statystyk
result_df = pd.DataFrame.from_dict(stats, orient='index', columns=['liczba cakowita', 'przeyli liczba', 'przeyli %', 'zginel liczba', 'zgineli %'])

# Clean and prepare data
df['pclass'] = df['pclass'].fillna(0)
df['age'] = df['age'].fillna(-1)
df['sibsp'] = df['sibsp'].fillna(0)
df['parch'] = df['parch'].fillna(0)
df['boat'] = df['boat'].notna().astype(int)
df['sex'] = df['sex'].fillna('unknown')

# Define categories
categories = {
    'pclass 1': (df['pclass'] == 1),
    'pclass 2': (df['pclass'] == 2),
    'pclass 3': (df['pclass'] == 3),
    'boat/numer': (df['boat'] == 1),
    'boat/NaN': (df['boat'] == 0),
    'male': (df['sex'] == 'male'),
    'female': (df['sex'] == 'female'),
    'sibsp/liczba': (df['sibsp'] > 0),
    'sibsp/0': (df['sibsp'] == 0),
    'parch/liczba': (df['parch'] > 0),
    'parch/0': (df['parch'] == 0),
    'age do 10': (df['age'] <= 10) & (df['age'] >= 0),
    'age>10 do 20': (df['age'] > 10) & (df['age'] <= 20),
    'age>20 do 40': (df['age'] > 20) & (df['age'] <= 40),
    'age>40 do 60': (df['age'] > 40) & (df['age'] <= 60),
    'age>60': (df['age'] > 60)
}

# Calculate survival rates
survival_rates = {}
for category, condition in categories.items():
    total = df[condition].shape[0]
    survived = df[condition & (df['survived'] == 1)].shape[0]
    if total > 0:
        survival_rate = (survived / total) * 100
    else:
        survival_rate = 0
    survival_rates[category] = survival_rate

# Select top 6 categories with highest survival rates
top_6_categories = sorted(survival_rates, key=survival_rates.get, reverse=True)[:6]
top_6_values = [survival_rates[cat] for cat in top_6_categories]

# Plotting
plt.figure(figsize=(10, 6))
plt.bar(top_6_categories, top_6_values, color='green')
plt.xlabel('Parametry')
plt.ylabel('Przeywalno (%)')
plt.title('Top 6 parametrw z najwyszym % przeywalnoci')
plt.ylim(0, 100)
plt.yticks(np.arange(0, 101, 10))
plt.xticks(rotation=45)
plt.tight_layout()

    'sibsp/liczba': df['sibsp'] > 0,
    'sibsp/0': df['sibsp'] == 0,
    'parch/liczba': df['parch'] > 0,
    'parch/0': df['parch'] == 0,
    'age do 10': df['age'] <= 10,
    'age>10 do 20': (df['age'] > 10) & (df['age'] <= 20),
    'age>20 do 40': (df['age'] > 20) & (df['age'] <= 40),
    'age>40 do 60': (df['age'] > 40) & (df['age'] <= 60),
    'age>60': df['age'] > 60
}

# Calculate mortality rates
mortality_rates = {}
for param, condition in conditions.items():
    total = df[condition].shape[0]
    if total > 0:
        deaths = df[condition & (df['survived'] == 0)].shape[0]
        mortality_rate = (deaths / total) * 100
        mortality_rates[param] = mortality_rate

# Select the top 6 parameters with the highest mortality rates
top_6_mortality = dict(sorted(mortality_rates.items(), key=lambda item: item[1], reverse=True)[:6])

# Plotting
plt.figure(figsize=(10, 6))
plt.bar(top_6_mortality.keys(), top_6_mortality.values(), color='red')
plt.xlabel('Parametry')
plt.ylabel('miertelno (%)')
plt.title('Top 6 parametrw z najwyszym % miertelnoci')
plt.ylim(0, 100)
plt.yticks(np.arange(0, 101, 10))
plt.xticks(rotation=45, ha='right')
