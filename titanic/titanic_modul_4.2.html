EDA - Analiza danych z katastrofy Titanica¶

O Danych¶

Dane o pasażerach Titanica

Zbiór danych zawiera informacje o pasażerach RMS Titanic, który zatonął 15 kwietnia 1912 roku po zderzeniu z górą lodową. Dane obejmują takie atrybuty jak klasa podróży, wiek, płeć, liczba rodzeństwa/małżonków na pokładzie, liczba rodziców/dzieci na pokładzie, cena biletu oraz miejsce zaokrętowania.

Zbiór zawiera także informację o tym, czy pasażer przeżył katastrofę.

Titanic przewoził ponad 2,200 osób, z czego ponad 1,500 zginęło, co czyni tę katastrofę jedną z najbardziej tragicznych w historii morskiej.

Kolumny:

  • pclass - Klasa biletu
  • survived - Czy pasażer przeżył katastrofę
  • name - Imię i nazwisko pasażera
  • sex - Płeć pasażera
  • age - Wiek pasażera
  • sibsp - Liczba rodzeństwa/małżonków na pokładzie
  • parch - Liczba rodziców/dzieci na pokładzie
  • ticket - Numer biletu
  • fare - Cena biletu
  • cabin - Numer kabiny
  • embarked - Port, w którym pasażer wszedł na pokład (C = Cherbourg, Q = Queenstown, S = Southampton)
  • boat - Numer łodzi ratunkowej
  • body - Numer ciała (jeśli pasażer nie przeżył i ciało zostało odnalezione)
  • home.dest - Miejsce docelowe

# Badanie wpływu istotnych zmiennych na przeżywalność pasażerów

# Funkcja do obliczania statystyk przeżycia
def calculate_survival_stats(df, condition):
    total_count = df[condition].shape[0]
    survived_count = df[condition & (df['survived'] == 1)].shape[0]
    died_count = total_count - survived_count
    survived_percent = (survived_count / total_count) * 100 if total_count > 0 else 0
    died_percent = (died_count / total_count) * 100 if total_count > 0 else 0
    return [total_count, survived_count, survived_percent, died_count, died_percent]

# Zdefiniowane warunki dla każdego parametru
conditions = {
    'pclass 1': df['pclass'] == 1,
    'pclass 2': df['pclass'] == 2,
    'pclass 3': df['pclass'] == 3,
    'male': df['sex'] == 'male',
    'female': df['sex'] == 'female',
    'boat /numer': df['boat'].notna(),
    'boat/NaN': df['boat'].isna(),
    'sibsp 0': df['sibsp'] == 0,
    'sibsp liczba': df['sibsp'] > 0,
    'parch 0': df['parch'] == 0,
    'parch liczba': df['parch'] > 0,
    'age do 10': df['age'] <= 10,
    'age >10 do 20': (df['age'] > 10) & (df['age'] <= 20),
    'age >20 do 40': (df['age'] > 20) & (df['age'] <= 40),
    'age > 40 do 60': (df['age'] > 40) & (df['age'] <= 60),
    'age > 60': df['age'] > 60
}

# Obliczenie statystyk dla każdego warunku
stats = {param: calculate_survival_stats(df, cond) for param, cond in conditions.items()}

# Utworzenie DataFrame ze statystyk
result_df = pd.DataFrame.from_dict(stats, orient='index', columns=['liczba całkowita', 'przeżyli liczba', 'przeżyli %', 'zginel liczba', 'zgineli %'])

# Clean and prepare data
df['pclass'] = df['pclass'].fillna(0)
df['age'] = df['age'].fillna(-1)
df['sibsp'] = df['sibsp'].fillna(0)
df['parch'] = df['parch'].fillna(0)
df['boat'] = df['boat'].notna().astype(int)
df['sex'] = df['sex'].fillna('unknown')

# Define categories
categories = {
    'pclass 1': (df['pclass'] == 1),
    'pclass 2': (df['pclass'] == 2),
    'pclass 3': (df['pclass'] == 3),
    'boat/numer': (df['boat'] == 1),
    'boat/NaN': (df['boat'] == 0),
    'male': (df['sex'] == 'male'),
    'female': (df['sex'] == 'female'),
    'sibsp/liczba': (df['sibsp'] > 0),
    'sibsp/0': (df['sibsp'] == 0),
    'parch/liczba': (df['parch'] > 0),
    'parch/0': (df['parch'] == 0),
    'age do 10': (df['age'] <= 10) & (df['age'] >= 0),
    'age>10 do 20': (df['age'] > 10) & (df['age'] <= 20),
    'age>20 do 40': (df['age'] > 20) & (df['age'] <= 40),
    'age>40 do 60': (df['age'] > 40) & (df['age'] <= 60),
    'age>60': (df['age'] > 60)
}

# Calculate survival rates
survival_rates = {}
for category, condition in categories.items():
    total = df[condition].shape[0]
    survived = df[condition & (df['survived'] == 1)].shape[0]
    if total > 0:
        survival_rate = (survived / total) * 100
    else:
        survival_rate = 0
    survival_rates[category] = survival_rate

# Select top 6 categories with highest survival rates
top_6_categories = sorted(survival_rates, key=survival_rates.get, reverse=True)[:6]
top_6_values = [survival_rates[cat] for cat in top_6_categories]

# Plotting
plt.figure(figsize=(10, 6))
plt.bar(top_6_categories, top_6_values, color='green')
plt.xlabel('Parametry')

    'pclass 1': df['pclass'] == 1,
    'pclass 2': df['pclass'] == 2,
    'pclass 3': df['pclass'] == 3,
    'boat/numer': df['boat'] > 0,
    'boat/NaN': df['boat'].isna(),
    'male': df['sex'] == 'male',
    'female': df['sex'] == 'female',
    'sibsp/liczba': df['sibsp'] > 0,
    'sibsp/0': df['sibsp'] == 0,
    'parch/liczba': df['parch'] > 0,
    'parch/0': df['parch'] == 0,
    'age do 10': df['age'] <= 10,
    'age>10 do 20': (df['age'] > 10) & (df['age'] <= 20),
    'age>20 do 40': (df['age'] > 20) & (df['age'] <= 40),
    'age>40 do 60': (df['age'] > 40) & (df['age'] <= 60),
    'age>60': df['age'] > 60
}

# Calculate mortality rates
mortality_rates = {}
for param, condition in conditions.items():
    total = df[condition].shape[0]
    if total > 0:
        deaths = df[condition & (df['survived'] == 0)].shape[0]
        mortality_rate = (deaths / total) * 100
        mortality_rates[param] = mortality_rate
